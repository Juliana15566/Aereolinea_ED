public List<Pasajero> buscarPorNombre(String nombre) {
    List<Pasajero> resultados = new ArrayList<>();

    String textoBuscado = nombre.toLowerCase().trim();

    for (Pasajero p : listaPasajeros) {
        String nombrePasajero = p.getNombre().toLowerCase();

        // Contiene el texto buscado
        if (nombrePasajero.contains(textoBuscado)) {
            resultados.add(p);
        }
    }

    // Mostrar resultados
    if (resultados.isEmpty()) {
        System.out.println("❌ No se encontraron pasajeros con: " + nombre);
    } else {
        System.out.println("✔ Pasajeros encontrados con el nombre '" + nombre + "':");
        for (Pasajero p : resultados) {
            System.out.println(p);  // Muestra toda la info del pasajero
        }
    }

    return resultados;
}
